Method 1: User Data Automation

User Data: Scripts that run automatically when EC2 instance starts

#!/bin/bash
# Update system
yum update -y

# Install Apache
yum install -y httpd

# Start Apache
systemctl start httpd
systemctl enable httpd

# Create web page
cat > /var/www/html/index.html << 'EOF'
<!DOCTYPE html>
<html>
<head>
    <title>My AWS Web Server</title>
</head>
<body>
    <h1>Welcome to My AWS Web Server!</h1>
    <p>This server was created automatically using User Data.</p>
</body>
</html>
EOF

Method 2: Manual Setup

Step-by-step approach: Connect to instance and configure manually

# 1. Connect to instance
ssh -i "your-key.pem" ec2-user@your-instance-ip

# 2. Update system
sudo yum update -y

# 3. Install Apache
sudo yum install -y httpd

# 4. Start Apache
sudo systemctl start httpd
sudo systemctl enable httpd

# 5. Create web page
sudo tee /var/www/html/index.html > /dev/null << 'EOF'
<!DOCTYPE html>
<html>
<head>
    <title>My Manual Web Server</title>
</head>
<body>
    <h1>Manually Configured Web Server!</h1>
    <p>This server was set up step by step.</p>
</body>
</html>
EOF

AWS CLI Commands

# Create security group
aws ec2 create-security-group \
    --group-name web-server-sg \
    --description "Security group for web server"

# Allow HTTP traffic
aws ec2 authorize-security-group-ingress \
    --group-name web-server-sg \
    --protocol tcp --port 80 --cidr 0.0.0.0/0

# Allow SSH traffic
aws ec2 authorize-security-group-ingress \
    --group-name web-server-sg \
    --protocol tcp --port 22 --cidr 0.0.0.0/0

# Launch instance with user data
aws ec2 run-instances \
    --image-id ami-0abcdef1234567890 \
    --count 1 \
    --instance-type t2.micro \
    --key-name your-key-pair \
    --security-groups web-server-sg \
    --user-data file://userdata.sh

Security Best Practices

üîí Security Groups

üîë Key Management
